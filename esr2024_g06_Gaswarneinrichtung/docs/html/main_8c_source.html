188 clk += intClockTime;
189 clkLEDAlert += intClockTime;
190 clkLEDOn += intClockTime;
191
192 //< On-Btn

198 //< Gas-Msrmnt

205 setAlertLED(false);

207 clkLEDAlert = 0;
208 setAlertLED(true);
209 }
210 }
211 } else if(warningState){
212 if(clkLEDAlert >= timeToBlinkLEDWarning){
213 setAlertLED(false);

220 setAlertLED(false);

255void interruptP3Handler(){
256 if(!warningState){
257 if(interruptP3p1){
258 timeWhenLastMoved = clk;
259 timeWhenLastCriticalMeasurement = clk;
260 interruptP3p1 = false;
261 warningState = true;
262 return;
263 }
264 return;
265 }
266 if(GPIO_getInputPinValue(GPIO_PORT_P3, GPIO_PIN1) == GPIO_INPUT_PIN_HIGH){
267 if(clk >= timeWhenLastCriticalMeasurement + timeWaitingAfterCriticalMeasurement
268 && !alarmState)
269 warningState = false;
270
271 }
272 if(interruptP3p5){
273 timeWhenLastMoved = clk;
274 interruptP3p5 = false;
275 return;
276 }
277
278 if(timeWhenLastMoved == 0){
279 timeWhenLastMoved = clk;
280 return;
281 }
282 if(clk > timeWhenLastMoved + timeRestingTillAlarm)
283 alarmState = true;
284}

292void interruptP4Handler(){

294 GPIO_enableInterrupt(GPIO_PORT_S1, GPIO_PIN_S1);

303 interruptP4 = false;
