179 clk += intClockTime;
180 clkLEDAlert += intClockTime;
181 clkLEDOn += intClockTime;

194 if(alarmState){
195 if(clkLEDAlert >= timeToBlinkLEDAlarm){
196 setAlertLED(false);

198 clkLEDAlert = 0;
199 setAlertLED(true);
200 }
201 }
202 } else if(warningState){
203 if(clkLEDAlert >= timeToBlinkLEDWarning){
204 setAlertLED(false);

206 clkLEDAlert = 0;
207 setAlertLED(true);
208 }

211 setAlertLED(false);

246void interruptP3Handler(){
247 if(!warningState){
248 if(interruptP3p1){
249 timeWhenLastMoved = clk;
250 timeWhenLastCriticalMeasurement = clk;
251 interruptP3p1 = false;
252 warningState = true;
253 return;
254 }
255 return;
256 }
257 if(GPIO_getInputPinValue(GPIO_PORT_P3, GPIO_PIN1) == GPIO_INPUT_PIN_HIGH){
258 if(clk >= timeWhenLastCriticalMeasurement + timeWaitingAfterCriticalMeasurement
259 && !alarmState)
260 warningState = false;
261
262 }
263 if(interruptP3p5){
264 timeWhenLastMoved = clk;
265 interruptP3p5 = false;
266 return;
267 }
268
269 if(timeWhenLastMoved == 0){
270 timeWhenLastMoved = clk;
271 return;
272 }
273 if(clk > timeWhenLastMoved + timeRestingTillAlarm)
274 alarmState = true;
275}

283void interruptP4Handler(){
284 if(!interruptP4){
285 GPIO_enableInterrupt(GPIO_PORT_S1, GPIO_PIN_S1);
286 return;
287 }
288
289 if(timeWhenOnBtnPressed == 0){
290 timeWhenOnBtnPressed = clk;
291 return;
292 }
293 if(GPIO_getInputPinValue(GPIO_PORT_S1, GPIO_PIN_S1) == GPIO_INPUT_PIN_HIGH){
294 interruptP4 = false;
295 return;
296 }
297 if(clk < timeWhenOnBtnPressed + timeToPressOnBtn) return; //2.5s
298
299
300 onState = !onState;
301 setBurnerActive(onState);
302 GPIO_disableInterrupt(GPIO_PORT_S1, GPIO_PIN_S1);
303 interruptP4 = false;
304 timeWhenOnBtnPressed = 0;
